#VERTEX_SHADER

#version 330 core

layout(location = 0) in vec4 position;

uniform mat4 view;
uniform mat4 projection;
uniform mat4 transform;

void main(){
	gl_Position = projection * view * transform * position;
}


#FRAGMENT_SHADER

#version 330 core

out vec4 FragColor;

void main(){
	FragColor = vec4(0.1, 0.1, 1.0, 0.7);
}
